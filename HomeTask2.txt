1.Daniel has 3 apples; Amber has 2 apples. How many apples do Daniel and Amber have together?

package task;

public class DanielAmber1 {

	public static void main(String[] args) {
	int daniel=3;
	int amber = 2;
	System.out.println("Daniel and Amber together have "+(daniel+amber)+ " apples");

	}
}
