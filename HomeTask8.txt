Write a program that analyzes the contents of the current directory and presents it as a collection objects in the form


package com.log4j.hometask;

import java.io.File;

public class DirectoryandSubdirectory {
	static void RecursivePrint(File[] arr, int index, int level) {
		// terminate condition
		if (index == arr.length)
			return;

		// tabs for internal levels
		for (int i = 0; i < level; i++)
			System.out.print("\t");

		// for files
		if (arr[index].isFile())
			System.out.println(arr[index].getName());

		// for sub-directories
		else if (arr[index].isDirectory()) {
			System.out.println("[" + arr[index].getName() + "]");

			// recursion for sub-directories
			RecursivePrint(arr[index].listFiles(), 0, level + 1);
		}

		// recursion for main directory
		RecursivePrint(arr, ++index, level);
	}

	// Driver Method
	public static void main(String[] args) {
		// Full path of actual directory
		String maindirpath = "D:\\Java\\Practice";

		File maindir = new File(maindirpath);

		if (maindir.exists() && maindir.isDirectory()) {

			File arr[] = maindir.listFiles();

			System.out.println("---------------------------------------");
			System.out.println("Files from main directory : " + maindir);
			System.out.println("---------------------------------------");

			// Calling recursive method
			RecursivePrint(arr, 0, 0);
		}
	}
}
