Question:
Write a program that creates an HTML table with the contents of the current directory in tabular format with below details
Name, Type, Creation Date, Size (in Kb)
--------------------------------------------------------

package com.iostream.htmltable;

import java.io.File;
import java.io.IOException;
import java.io.PrintWriter;
import java.io.StringWriter;
import java.sql.Date;

import javax.swing.JEditorPane;
import javax.swing.JFrame;
import javax.swing.JScrollPane;

public class CreateHTMLTable {
	public static void main(String[] args) throws IOException {
		String dirname = "D:/Unix";

		final JEditorPane editor = new JEditorPane();
		editor.setEditable(false);
		editor.setContentType("text/html");
		editor.setText(makeHTMLTable(dirname));

		JFrame frm = new JFrame("HTML Table");
		frm.getContentPane().add(new JScrollPane(editor));
		frm.setSize(800, 500);
		frm.setVisible(true);
	}

	public static String makeHTMLTable(String dirname) {
		File dr = new File(dirname);
		String[] entries = dr.list();

		StringWriter sw = new StringWriter();
		PrintWriter pw = new PrintWriter(sw);

		pw.println("<H1>" + dirname + "</H1>");
		pw.print("<TABLE BORDER=2 WIDTH=800><TR>");
		pw.print("<TH>Name</TH><TH>File Type</TH><TH>Size</TH><TH>Date Created</TH>");

		for (int i = 0; i < entries.length; i++) {
			File f = new File(dr, entries[i]);
			pw.println("<TR><TD>" + (f.isDirectory() ? entries[i] : entries[i])
					+ "</TD><TD>" + (f.isDirectory() ? "DIR" : "FILE")
					+ "</TD><TD>" + f.length() + "</TD><TD>"
					+ new Date(f.lastModified()) + "</TD></TR>");
		}

		pw.println("</TABLE>");
		pw.close();
		return sw.toString();
	}
}
